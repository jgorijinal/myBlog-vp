class myRequest {
  constructor(BASE_URL){
    this.BASE_URL = BASE_URL
  }
  request(options) {
    return new Promise((resolve, reject) => {
      wx.request({
        ...options,
        url: this.BASE_URL + options.url
        success: (res)=>{
          resolve(res.data)
        },
        fail:(err)=>{
          reject(err)
        }
      })
    })
  }
  get(options) {
    return this.request({...options , method: 'get'})
  }
  post(options) {
    return this.request({...options , method: 'post'})
  }
}

export const reqInstance = new myRequest('http://123.207.32.32:1888/api')

# 网络请求域名配置

每个微信小程序需要事先设置通讯域名，小程序只可以跟指定的域名进行网络通信。

  • 小程序登录后台 – 开发管理 – 开发设置 – 服务器域名；

服务器域名请在 「小程序后台 - 开发 - 开发设置 - 服务器域名」 中进行配置，配置时需要注意：

  • 域名只支持 https (wx.request、wx.uploadFile、wx.downloadFile) 和 wss (wx.connectSocket) 协议；
  • 域名不能使用 IP 地址（小程序的局域网 IP 除外）或 localhost；
  • 可以配置端口，如 https://myserver.com:8080，但是配置后只能向 https://myserver.com:8080 发起请求。如果向 https://myserver.com、https://myserver.com:9091 等 URL 请求则会失败。
  • 如果不配置端口。如 https://myserver.com，那么请求的 URL 中也不能包含端口，甚至是默认的 443 端口也不可以。如果 向 https://myserver.com:443 请求则会失败。
  • 域名必须经过 ICP 备案；
  • 出于安全考虑，api.weixin.qq.com 不能被配置为服务器域名，相关 API 也不能在小程序内调用。 开发者应将 AppSecret 保存到后台服务器中，通过服务器使用 getAccessToken 接口获取 access_token，并调用相关 API；
  • 不支持配置父域名，使用子域名

# 分享功能

分享是小程序扩散的一种重要方式，小程序中有两种分享方式：

  • 方式一：点击右上角的菜单按钮，之后点击转发
  • 方式二：点击某一个按钮，直接转发

当转发给好友一个小程序时，通常小程序中会显示一些信息：

  • 如何决定这些信息的展示呢？通过 onShareAppMessageopen in new window
  • 监听用户点击页面内转发按钮（button 组件 open-type="share"）或右上角菜单“转发”按钮的行为，并自定义转发内容。
  • 此事件处理函数需要 return 一个 Object，用于自定义转发内容； 
Page({
  // 分享

    return {
      title:'标题',
      path:'/pages/logs/logs',
      imageUrl:'/assets/123.jpg'

  }

# 小程序登录解析

如何识别同一个小程序用户身份？

  • 认识小程序登录流程
  • openid 和 unionid
    • openid : 相当于我们在微信登录的过程当中, 它的身份的唯一标识
    • unionid: 在微信的不同产品之间, 想要识别是同一个用户身 份的时候, 所用到的标识符
  • 获取 code
    • 需要用 code 换取 openid
  • 换取 authToken

用户身份在多个平台进行共享(比如 PC端 和 小程序)

  • 账号绑定 或者
  • 手机号登录
  • 目的: 目的是把 openid 和账号或者手机号进行关联, 进行多平台的共享

# 小程序用户登录的流程

对于小程序来说, 其实需要做的事情只有两个操作, 第一个操作: 拿到 code , 把这个 code 发送给服务器, 第二个操作: 后端会返回 token , 拿到这个 token 并把 token 保存下来

Page({
  onLoad(){
    // 1. 判断 token , 判断 token 是否有值
    const token = wx.getStorageSync('token')
    // 2. 如果 token 有值, 
    if(token) {
      console.log('请求其他数据')
    } else {
      // 如果没有 token , 就登录
      this.handleLogin()
    }
  },
  handleLogin(){
        // 1. 获取 code
        wx.login({
          success:(res) => {
            const code = res.code
            // 2. 将这个 code 发送给自己的服务端(后端)
            wx.request({
              url:'http://123.207.32.32:3000/login',
              data: {
                code
              },
              method: 'post',
              success(res){
                const token =  res.token
                wx.setStorageSync('token', data)
              }
            })
        }
    })
  }
})

在页面中: (关于网络请求 APi 在前面已经封装过, 这里不在展开)

import { getCode} from '../../service/login'
Page({
  onLoad(){
    // 1. 判断 token , 判断 token 是否有值
    const token = wx.getStorageSync('token') || ""

    // 2. 判断 token 是否过期
    const res = await myRequest.post({
      url:'/auth',
      header:{
        token: token
      }
    })

    // 3. 如果 token 有值, 并没有过期
    if(token && res.message === '已登录') {
      console.log('请求其他数据')
    } else {
      // 如果没有 token , 就登录
      this.handleLogin()
    }
  },
  async handleLogin(){   // 微信登录操作
    // 1. 获取 code 
    const code = await getCode()
    // 2. 使用 code 换取 token
    const res = await myRequest.post({
      url:'/login',
      data: { code }
    })
    // 3. 保存 token 
    wx.setStorageSync('token', res.token) 
  }
})
