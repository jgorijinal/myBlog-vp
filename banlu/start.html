创建 utils/request.js

module.exports = (vm) => {
    // 初始化请求配置
    uni.$u.http.setConfig((config) => {
        /* config 为默认全局配置*/
        config.baseURL = 'http://uat.banlu.xuexiluxian.cn';
        return config
    })
	// 请求拦截
	uni.$u.http.interceptors.request.use((config) => { 
	    return config 
	}, config => { 
	    return Promise.reject(config)
	})
	
	// 响应拦截
	uni.$u.http.interceptors.response.use((response) => {
		const data = response.data
		return data.data === undefined ? {} : data.data
	}, (response) => { 
		return Promise.reject(response)
	})
}

在混入 mixin 里面封装一个 getHeight 计算高度的方法,因为获取高度的方法在其他页面中经常会用到。 它的作用是 getHeight(selector1 , selector2) 在参数里可以传入选择器计算出除了这些选择器之外的页面的可用高度, 当然也可以不传参数

mixin/mixins.js

export default {
  methods: {
      getHeight(selector1 , selector2){ // 获取可用区域的高度
        let height = 0
        let s1Height = 0
        let s2Height = 0
        if(selector1) {
          const query = uni.createSelectorQuery()
          query.select(selector1).boundingClientRect(data =>{
            s1Height = data.height
          }).exec()
        }
        if(selector2) {
          const query = uni.createSelectorQuery()
          query.select(selector2).boundingClientRect(data =>{
            s2Height = data.height
          }).exec()
        }
        uni.getSystemInfo({
          success(res) {
            height = res.windowHeight - s1Height - s2Height
          }
        })
        return height
      }
  }
}

# H5端跨域处理

根据官方文档的描述，devServer配置被要求在manifest.json去配置，并且由于这个配置文件是json格式的，所以只能对简单类型进行配置。但对于proxy这项配置来说也是足够了的。直接如下方式配置即可解决:

# 另一种解决方案

直接创建一个 vue.config.js 文件，并在里面配置 devServer，直接上代码

module.exports = {
	devServer:{
		proxy:{
			'/':{
				target:'http://uat.banlu.xuexiluxian.cn',
				changeOrigin:true
			}
		}
	}
}

# 触发数据请求

在 onLoad 生命周期调用接口

在首页:

<template>
	<view class="content">
    <swiper :indicator-dots="false" :vertical="true" :autoplay="true" :interval="3000" :duration="1000" 
      :style="'height:'+swiperHeight+'px;'">
      <swiper-item class="item" v-for="item,index in goodsList" :key="index">
        <image class="image" :src="item.skus[0].resources[0].resPath">
          <view class='shopping'>
          	<view class='shop-cart'>
          		<image src="../../static/images/shop.png" mode=""></image>
          	</view>
          	<view class='shop-title'>
          		<view>{{item.title}}</view>
          	</view>
          </view>
      </swiper-item>
    </swiper>
	</view>
</template>

<script>
  import mixins from '@/mixin/mixins.js'
  import { getGoodsPage } from  '@/utils/api/goods.js'
	export default {
    mixins:[mixins],
		data() {
			return {
        swiperHeight:0, // 可滚动高度
        goodsList: []
			}
		},
		onLoad() {
      getGoodsPage({ // 发请求获取商品列表
        current:1,
        size: 20
      }).then(res =>{
        res.records.forEach(record =>{ // 后端返回给前端的数据是字符串的对象, 所以使用JSON反序列化把数据转为普通对象 
          record.skus[0].resources = JSON.parse(record.skus[0].resources)
        })
        this.goodsList = res.records
      })      
		},
    onReady(){
      this.swiperHeight = this.getHeight()  // geHeight 不需要传入参数
    },
	}
</script>
