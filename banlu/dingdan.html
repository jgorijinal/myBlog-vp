# 订单列表

<template>
  <view class="order">
    <u-subsection :list="list" mode="button" :current="current" @change="tabChange"></u-subsection>
    <view class="order-lists">   
    			<view 
    				v-for="(item,index) in orderList" 
    				:key='index'
    				@click='orderDetail(item.id)'
    			>
    				<view class="order-list">
    					<view class="list-top">
    						<view class="list-img">
    							<image :src="item.goodsList[0].resources[0].resPath"></image>
    						</view>
    						<view class="list-info">
    							<view class="list-title double-hidden">{{item.goodsList[0].spuTitle}}</view>
    							<view class="list-brief single-hidden">{{item.goodsList[0].skuTitle}}</view>
    						</view>
    						<view class="list-price">
    							<view class="price-icon">￥</view>
    							<view class="price-num">{{item.payPrice}}</view>
    						</view>
    					</view>
    					<view class="list-bottom">
    						<view class="pay">
    							<view> 实付：</view>
    							<view class="total-price">
    								<view class="price-icon">￥</view>
    								<view class="price-num">{{item.payPrice}}</view>
    							</view>
    						</view>           
    						<view class="order-status" v-if="item.status == 1 && item.deliveryStatus == 0 && item.isPay == 2"> 
    							<view class="show-status receipt">待付款</view> 
    						</view>
    						<view class="order-status" v-else-if="item.status==1 && item.deliveryStatus==0 && item.isPay==1"> 
    							<view class="show-status receipt">待发货</view> 
    						</view>
    						<view class="order-status" v-else-if="item.status==1 && item.deliveryStatus==1"> 
    							<view class="show-status receipt">待收货</view>        
    						</view>
    						<view class="order-status" v-else-if="item.status == 2 || item.status == 3">
    							<view class="show-status receipt">已关闭</view> 
    						</view>                  
    						<view class="order-status" v-else-if="item.status == 4">
    							<view class="show-status receipt">已退款</view> 
    						</view>
    						<view class="order-status" v-else-if="item.status == 5 && item.deliveryStatus == 2">
    							<view v-if="item.isReturns === 0" class="show-status after-sales">申请售后</view>
    							<view class="show-status finished">已完成</view>
    						</view>
    					</view>
    				</view>
    			</view>
    		</view>
  </view>
</template>

<script>
  import { getOrderList } from '@/utils/api/order.js'
  export default {
    data(){
      return {
       list: ['全部','待付款','待发货','待收货','已完成'],
       current: 0,  // tab的索引
       params: {  // 获取订单列表参数
           current:'1',
           size:'10',
           isPay: '', // 是否支付（1：已支付；2：未支付）
           deliveryStatus:'', // 发货状态 - 0：未发货；1：已发货；2：已收货
           status:'' // 	状态 - 1：正常；2：超时；3：关闭；4：退款；5：完成
       },
       orderList: [] // 订单列表
      }
    },
    onLoad(options){
     this.tabChange(Number(options.active))
    },
    methods: {
      getOrderList(){  // 湖区订单列表(单独封装)
        console.log(this.params)
        getOrderList(this.params).then(res => {
          res.data.data.records.forEach(item => {
            item.goodsList.forEach(obj =>  {
                obj.resources = JSON.parse(obj.resources)
            })
          })
          this.orderList = res.data.data.records
          console.log(res.data)
        })
      }, 
      tabChange(index){
        this.current = index
       if(index === 0) { // 全部
         this.getOrderList(this.params)
       }else if(index === 1){ // 待付款
        this.params.isPay = '2'
        this.params.deliveryStatus = '0'
        this.params.status = '1'
        this.getOrderList(this.params)
       } else if(index === 2 ) { // 待发货
         this.params.isPay = '1'
         this.params.deliveryStatus = '0'
         this.params.status = '1'
         this.getOrderList(this.params)
       } else if(index === 3) {  // 待收货
         this.params.isPay = '1'
         this.params.deliveryStatus = '1'
         this.params.status = '1'
         this.getOrderList(this.params)
       } else if (index === 4) { // 已完成
          this.params.isPay = '1'
          this.params.deliveryStatus = '2'
           this.params.status = '5';
          this.getOrderList(this.params);
       }
      },
      orderDetail(id) {
        console.log(id)
      }
    }
  }
</script>
