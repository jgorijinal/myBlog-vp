# 持久化存储 (Vuex-persistedstate 插件)

使用 vuex-persistedstate 插件来做本地持久化存储,

vuex-persistedstate：可以自动保存 Vuex 中的数据到 本地存储，并且在下次开始的时候自动读取这个数据到对应的 state 中

  1. 安装 vuex-persistedstate 插件
npm install vuex-persistedstate
1
  1. 在 src/store/index.js 中导入，并注册 plugin
import Vue from 'vue'
import Vuex from 'vuex'
import user from './modules/user.js'
import  createPersistedState from 'vuex-persistedstate'
Vue.use(Vuex)
const store = new Vuex.Store({
  state:{
  },
  modules:{
    user
  },
  plugins: [
    createPersistedState({
      key: 'uni-project',
      storage:{
        getItem: (key) => uni.getStorageSync(key), //获取,
        setItem: (key,value) => uni.setStorageSync(key,value), //存储,
        removeItem: (key) => uni.removeStorageSync(key), //删除,
      }
    })
  ]
})

export default store

# 微信一键登录

# ! 微信一键登录流程 !

  1. 调用官方 uni.getUserProfile(obj)open in new window 接口获取用户信息
  2. 调用官方 uni.login(obj)open in new window 接口获取临时登录凭证code
  3. 当前平台需要验证一下该用户是否在此平台注册过, 前端调用后台接口{ 微信一键登录 }传递 code码
  • 如果注册过, 后端会返回 200 / 未注册过会返回 60003 的 code(状态码)
  • 未注册过, 即 60003:
    • (1) 如果该用户未注册过, 需要调接口注册一下 { 注册微信用户 }, 并且需要传递参数(看接口文档)
    • (2) 调接口后, 在该平台上注册成功, 后端会返回 200 状态码, 然后走下面流程 (判断绑定手机逻辑)
  • 已注册过, 即 200:
    • (1) 存储 token
    • (2) 判断用户是否绑定过手机号, 即判断一下 res.data.member.mobile 存不存在
    • 如果不存在, 跳转到 -> 绑定手机号的页面
    • 如果存在, 返回到 "我的" 页面
<template>
  <view class="login">
    <button type="primary" @click="wxLogin">微信一键登录</button>
  </view>
</template>

<script>
  import {loginByWechat, registerWechat} from  '@/utils/api/login.js'
  import store from '@/store/index.js'
  export default {
    data(){
      return {
        registerData : {}
      }
    },
    methods:{
      // 点击微信一键登录
      wxLogin(){
        let that = this
        uni.getUserProfile({  //  uni.getUserProfile()
          desc:'登录后同步数据',
          success(res1){
            uni.login({  
              success(res2){
                 const params = {
                   code: res2.code
                 }
                 loginByWechat(params).then(res3 => {
                   if(res3.data.code  === "60003") { // 未注册
                     that.registerData = {  // 注册微信用户所需要的参数
                       unionId: res3.data.unionId ? res3.data.unionId : res3.data.data.openid,
                       openId:res3.data.data.openid,
                       sessionKey:res3.data.data.sessionKey,
                       signature:res1.signature,
                       rawData:res1.rawData,
                       encryptedData:res1.encryptedData,
                       iv:res1.iv
                     }
                     that.register()
                   }else if(res3.data.code  === "200") { //已注册 (有 token)
                      that.checkBindMobile(res3.data.data)
                   }
                 })
              }
            })
          }
        })
      },
      // 微信用户注册
      register(){
        registerWechat(this.registerData).then(res => {
          if(res.data.code === '200') {
            console.log('注册成功')
            this.checkBindMobile(res.data.data)   // 检查是否绑定过手机号
          }else {
            console.log('注册失败')
          }
        })
      },
      // 检查是否绑定过手机号
      checkBindMobile(data){
        // 存储 token
        // uni.setStorageSync('token', data.token)
        store.commit('setToken', data.token)                //  <---------------存 token
        if(!data.member.mobile) {
          // 没有查到手机号 
          // 跳转到手机绑定页面 , 去绑定手机
         this.showBindMobilePage()
        } else { // 绑定过手机
          // 跳转到我的页面
          uni.switchTab({
            url:'/pages/tabs/me'
          })
           uni.showToast({
            title:'登录成功'
          })
        }
      },
      // 跳转到绑定手机号页面
      showBindMobilePage(){
        uni.showModal({ // 对话框
          title: '友情提示',
          content: '根据国家的规定, 需要绑定手机号, 是否要绑定手机号? ',
          confirmText:'去完善',
          success(res){
            if(res.confirm) { // 点击了确定
                uni.navigateTo({
                  url:'/pages/login/mobile'
                })
            }else if(res.cancel) { // 点击了取消
              // TODO :如果不绑定手机号的话 , 删除 token 
              console.log('点击了取消')
            }
          },
        })
      }
    }
  }
</script>

# 请求拦截统一注入 token

判断 Vuex中存储的 token 存不存在, 如果 token 存在就说明用户已经是登录状态 -> 在请求拦截器中请求头统一注入 tokenutils/request.js

// 此vm参数为页面的实例，可以通过它引用vuex中的变量
module.exports = (vm) => {
    // 初始化请求配置
    uni.$u.http.setConfig((config) => {
        /* config 为默认全局配置*/
        // #ifdef MP-WEIXIN
        config.baseURL = 'http://uat.banlu.xuexiluxian.cn';
        // #endif
        return config
    })
	// 请求拦截
	uni.$u.http.interceptors.request.use((config) => { 
    	//判断token是否存在，如果存在传递给后端
      if(store.state.user.token){
        config.header = {
          'content-type': 'application/json',
          'Authorization': store.state.user.token
        }
      }
	    return config 
	}, config => { 
	    return Promise.reject(config)
	})
	
	// 响应拦截
	uni.$u.http.interceptors.response.use((response) => {
		return response
	}, (response) => { 
		return Promise.reject(response)
	})
}

# 绑定手机号 (加密处理)

现已跳转到了绑定手机号的页面

这里需要使用 Aes 对称加密: 给明文加密

npm install crypto-js -S

utils/aes.js

// 引入AES源码js ==> npm install crypto-js -S
const CryptoJS = require('crypo-js')

// 默认的KEY与IV如果没有给
// 秘钥：bGvnMc62sh5RV6zP
// 偏移量：1eZ43DLcYtV2xb3Y
const key = CryptoJS.enc.Utf8.parse('bGvnMc62sh5RV6zP') // 十六位十六进制数作为密钥
const iv = CryptoJS.enc.Utf8.parse('1eZ43DLcYtV2xb3Y') // 十六位十六进制数作为密钥偏移量

// 解密方法
export function Decrypt(word) {
    const encryptedHexStr = CryptoJS.enc.Hex.parse(word)
    const srcs = CryptoJS.enc.Base64.stringify(encryptedHexStr)
    const decrypt = CryptoJS.AES.decrypt(srcs, key, { iv: iv, mode: CryptoJS.mode.CBC, padding: CryptoJS.pad.Pkcs7 })
    const decryptedStr = decrypt.toString(CryptoJS.enc.Utf8)
    return decryptedStr.toString()
}

// 加密方法
export function Encrypt(word) {
    const srcs = CryptoJS.enc.Utf8.parse(word)
    const encrypted = CryptoJS.AES.encrypt(srcs, key, { iv: iv, mode: CryptoJS.mode.CBC, padding: CryptoJS.pad.Pkcs7 })
    return encrypted.ciphertext.toString().toUpperCase()
}

  <view class="verification-btn" @click='sendCode'>
      <button type="primary" class="verification-code"  :disabled="isShowCountDown">
        <u-count-down :time="60 * 1000" format="ss" v-if="isShowCountDown" @finish="isShowCountDown = false"></u-count-down>
        <text v-else>获取验证码</text>
      </button>
  </view>
</template>
<script>
export default {
  ...
  data(){
    return {
      phone: '', // 手机号
      code:'',  // 验证码
      isShowCountDown: false // 是否显示倒计时
    }
    },
   methods:{
      sendCode(){// 发送短信获取验证码
        const reg = /^1[3-9\d{9}$]/  // 手机号码正则校验      <-------------- 开始看
        if(!reg.test(this.phone)) {
          // 校验不通过
          uni.showToast({  // 弹窗提示用户
            title:'请输入正确格式的手机号!',
            icon:'none',
            duration: 1500
          })
        }else {
          // 手机校验通过
          let key = 'MODIFY_MOBILE'
          sendMessage({
            mobile:  Encrypt(this.phone),
            key
          }).then(res => {
            if(res.data.code === '200') {
                this.isShowCountDown = true    // 显示倒计时
            }
          })
        }
      },
    }
}
</script>

# 获取用户信息

获取并展示用户信息 把它分为三部分进行实现：

  1. 定义接口请求方法
  2. 定义调用接口的动作
  3. 在路由拦截时触发动作

那么接下来就根据这三个步骤，分别来进行实现：

定义接口请求方法：

// 用户相关
const http = uni.$u.http
....
// 获取用户信息
export const getUserInfo = (data) => {
  return http.get('/account/info', {params:data})
}

uView 的请求库在有 bug, 不能(因为会报错)再 store 的 actions 里面引入请求接口, 所以在 Vuex 这里只封装 mutations 对应的存储 userInfo的方法

store/modules/user.js

export default {
  state:{
    token: '', // token
    userInfo: {} // 用户信息
  },
  mutations:{
    setToken(state, token){
      state.token = token
    },
    setUserInfo(state, userInfo){
      state.userInfo = userInfo
    }
  }
}
