# 收藏

  1. 先定义请求接口

utils/api/goods.js

const http = uni.$u.http
...
// 收藏商品
export const addCollect = (id) =>{
  return http.get('/collect/add/' + id)
}
// 取消商品收藏
export const deleteCollect = (id) => {
  return http.get('/collect/delete/' + id)
}

逻辑:

  • 即使是调用了(收藏/取消收藏)接口并调用成功, 但本地的数据是没有被改变
  • 当前的数据写成了计算属性, 想要修改计算属性需要设置成 getter/setteropen in new window
  • 最后给按钮处理节流open in new window, 防止多次被频繁调用

goods-detail.vue

<template>
    <view class="popup">
        <image class="goods-image" :src="goodsItem.img" mode="widthFix"></image>
       </swiper>
       <view class="price">¥{{goodsItem.currentPrice}}</view>
       <view class="title">{{goodsItem.title}}</view>
       <view class="action">
         <!--分享-->
         <button class="share" open-type="share">
            <image class="icon" src="../../static/images/share1.png" mode="heightFix"></image>
            <text>分享</text>
         </button>
         <!--收藏-->
         <button class="like" @click="$u.throttle(onClickCollect, 2000, immediate = true)">
            <image v-if="isCollected === 0" class="icon" src="../../static/images/collection1.png" mode="heightFix"></image>
            <image v-else class="icon" src="../../static/images/collection2.png" mode="heightFix"></image>
            <text :class="isCollected === 1 ? 'isCollected' : '' ">收藏</text>
         </button>
         <!--立即购买-->
         <u-button color="#3d3d3d" shape="circle" style="margin-right:16px">立即购买</u-button>
       </view>
    </view>
</template>

<script>
  import {addCollect, deleteCollect} from '@/utils/api/goods.js'
  export default {
    name:"goods-detail",
    props: {
      item :{  // 商品的数据
        type:Object,
        default: () => {}
      }
    },
    computed:{
      goodsItem(){
        return {
          img:'http://' + this.item?.skus[0]?.resources[0]?.resPath,
          title:this.item?.title,
          currentPrice:this.item?.skus[0]?.currentPrice,
          id: this.item?.id  // 商品的 id
        }
      },
      isCollected:{  // 是否收藏 : 0 未收藏 , 1 已收藏  本地数据需要变化, 所以设置 getter/setter
        //getter
        get(){
          return this.item?.isCollected
        },
        //setter
        set(newValue) {
          this.item.isCollected = newValue
        }
      },
    },
    methods:{
      onClickCollect(){
        if(this.isCollected === 1) {
          // 已收藏, 要取消
          deleteCollect(this.goodsItem.id).then(res => {
            if(res.data.code === '200') {
                uni.showToast({
                  title:'已取消收藏!',
                  icon:'none'
                })
                this.isCollected = 0
            }
          })
        } else {
          // 未收藏, 要收藏
          addCollect(this.goodsItem.id).then(res =>{
           if(res.data.code === '200') {
             uni.showToast({
               title:'收藏成功!',
               icon:'none'
             })
               this.isCollected = 1
           }
          })
        }
      }
    }
  }
</script>
